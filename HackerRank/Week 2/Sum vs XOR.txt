#include <stdio.h>

int main() {
    unsigned long long n;
    scanf("%llu", &n);
    
    int zero_bits = 0;
    
    while (n) {
        if ((n & 1) == 0) {
            zero_bits++;
        }
        n >>= 1;
    }
    unsigned long long result = 1ULL<< zero_bits;
    printf("%llu\n", result);
    
    return 0;
}
