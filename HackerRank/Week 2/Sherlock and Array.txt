#include <stdio.h>
int main() {
    int T;
    scanf("%d", &T);
    
    while (T--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        long long total = 0, left_sum = 0;
        
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
            total += arr[i];
        }
        
        int found = 0;
        for (int i = 0; i < n; i++) {
            if (left_sum == total - left_sum - arr[i]) {
                found = 1;
                
                break;
            }
            left_sum += arr[i];
        }
        if (found)
            printf("YES\n");
        else
            printf("NO\n");    
    }
    return 0;
}
