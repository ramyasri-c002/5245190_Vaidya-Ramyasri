DoublyLinkedListNode* reverse(DoublyLinkedListNode* llist) {
if(llist == nullptr) return nullptr;
DoublyLinkedListNode* curr = llist;
DoublyLinkedListNode* temp = nullptr;
while (curr != nullptr){
    temp = curr ->prev;
    curr->prev = curr->next;
    curr->next = temp;
    curr = curr->prev;
    
}
if(temp != nullptr){
    llist = temp->prev;
}
return llist;
}
