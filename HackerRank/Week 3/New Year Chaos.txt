#include <stdio.h>

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int arr[n];
        for (int i = 0; i < n; i++)
            scanf("%d", &arr[i]);
            
            int chaotic = 0;
            int bribes = 0;
            
            for (int i = 0; i < n; i++) {
                if (arr[i] - (i + 1) > 2) {
                    chaotic = 1;
                    break;
                }
                
                for (int j = (arr[i] -2 > 0 ? arr[i] - 2 : 0); j < i; j++) {
                    if (arr[j] > arr[i]) bribes++;
                }
            }
            if (chaotic)
                printf("Too chaotic\n");
            else
                printf("%d\n", bribes);    
    }
    return 0;
}
