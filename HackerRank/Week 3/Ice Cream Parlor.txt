#include <stdio.h>

void whatFlavours(int cost[], int n, int money) {
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (cost[i] + cost[j] == money) {
                printf("%d %d\n", i + 1, j + 1);
                return;
            }
        }
    }
}

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int money, n;
        scanf("%d", &money);
        scanf("%d", &n);
        int cost[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &cost[i]);
        }
        whatFlavours(cost, n, money);
    }
    return 0;
}
