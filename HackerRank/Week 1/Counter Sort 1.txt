#include <stdio.h>

int main() {
    int n;
    if (scanf("%d", &n) != 1) return 0;
    int freq[100] = {0};
    for (int i = 0; i < n; i++) {
        int x;
        if(scanf("%d", &x) != 1) break;
        if (x >= 0 && x < 100) freq[x]++;
    }
    for (int i = 0; i < 100; i++) {
        if (i) putchar(' ');
        printf("%d", freq[i]);
    }
    putchar('\n');
    return 0;
}
