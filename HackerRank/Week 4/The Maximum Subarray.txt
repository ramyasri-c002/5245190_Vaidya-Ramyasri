#include <stdio.h>

int main() {
    int T;
    scanf("%d", &T);
    
    while (T--) {
        int n;
        scanf("%d", &n);
        
        int arr[n];
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
        }
        
        int max_sum = arr[0];
        int current_sum = arr[0];
        
        for (int i = 1; i < n; i++) {
            if (current_sum + arr[i] > arr[i])
                current_sum = current_sum + arr[i];
            else
                current_sum = arr[i];
                
            if (current_sum > max_sum)
                max_sum = current_sum;
        }
        
        int subseq_sum = 0;
        int has_positive = 0;
        int max_element = arr[0];
        
        for (int i = 0; i < n; i++) {
            if (arr[i] > 0) {
                subseq_sum += arr[i];
                has_positive = 1;
            }
            if (arr[i] > max_element)
                max_element = arr[i];
        }
        
        if (!has_positive) subseq_sum = max_element;
        
        printf("%d %d\n", max_sum, subseq_sum);
    }
    
    return 0;
}
